159 public static double getExceedance(ArrayList<Double> optedData, double exc) {
160 int size = optedData.size();
161 if (size==0){
162 Error.addEvaluationError("The data in the specified time period and selected month(s) from the timeseries doesn't exist in the Exceedance function.");
163 return 1.0;
164 }
165 Double[] values =optedData.toArray(new Double[size]);
166 Arrays.sort(values);
167 if (exc==1){
168 return values[0];
169 }
170 double excIndex = (1.0-exc)*size;
171 if (excIndex>=1.0){
172 int index = (int) Math.floor(excIndex);
173 double fract = excIndex-index;
174 double value = values[index-1]+fract*(values[index]-values[index-1]);
175 return value;
176 }else{
177 double value=values[0]-(1.0-excIndex)*(values[1]-values[0]);
