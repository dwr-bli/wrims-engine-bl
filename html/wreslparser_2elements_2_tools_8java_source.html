121 public static String readFileAsString(String file, String csName) throws IOException {
122 Charset cs = Charset.forName(csName);
123 // Thanks to Jon Skeet
124 // No real need to close the BufferedReader/InputStreamReader
125 // as they're only wrapping the stream
126 FileInputStream stream = new FileInputStream(file);
127 try {
128 Reader reader = new BufferedReader(new InputStreamReader(stream, cs));
129 StringBuilder builder = new StringBuilder();
130 char[] buffer = new char[8192];
131 int read;
132 while ((read = reader.read(buffer, 0, buffer.length)) > 0) {
133 builder.append(buffer, 0, read);
134 }
135 return builder.toString();
136 }
137 finally {
138 // Potential issue here: if this throws an IOException,
139 // it will mask any others. Normally I'd use a utility
140 // method which would log exceptions and swallow them
141 stream.close();
142 }
