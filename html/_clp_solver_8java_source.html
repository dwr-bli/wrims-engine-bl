362 private static void assignDvar() {
363 Map<String, Map<String, IntDouble>> varCycleValueMap=ControlData.currStudyDataSet.getVarCycleValueMap();
364 Map<String, Map<String, IntDouble>> varTimeArrayCycleValueMap=ControlData.currStudyDataSet.getVarTimeArrayCycleValueMap();
365 Set<String> dvarUsedByLaterCycle = ControlData.currModelDataSet.dvarUsedByLaterCycle;
366 Set<String> dvarTimeArrayUsedByLaterCycle = ControlData.currModelDataSet.dvarTimeArrayUsedByLaterCycle;
367 ArrayList<String> timeArrayDvList = ControlData.currModelDataSet.timeArrayDvList;
368 String model=ControlData.currCycleName;
369
370 StudyDataSet sds = ControlData.currStudyDataSet;
371 ArrayList<String> varCycleIndexList = sds.getVarCycleIndexList();
372 ArrayList<String> dvarTimeArrayCycleIndexList = sds.getDvarTimeArrayCycleIndexList();
373 Map<String, Map<String, IntDouble>> varCycleIndexValueMap = sds.getVarCycleIndexValueMap();
374
375 Map<String, Dvar> dvarMap = SolverData.getDvarMap();
376 Set dvarCollection = dvarMap.keySet();
377 Iterator dvarIterator = dvarCollection.iterator();
378
379 while(dvarIterator.hasNext()){
380 String dvName=(String)dvarIterator.next();
381 Dvar dvar=dvarMap.get(dvName);
382
383 double value = -77777777;
384 try {
385 if ( varDoubleMap.keySet().contains(dvName) ) {
386 value=varDoubleMap.get(dvName);
387 } else {
388 value = 0; // if not listed in clp solution file then it's 0
389 varDoubleMap.put(dvName, value);
390 }
391
392 } catch (Exception e) {
393 errorCode=12;
394 Error.addSolvingError("CLP solution dvar value error. Dvar: "+dvName+" Value: "+varDoubleMap.get(dvName));
395
396 }
397 IntDouble id=new IntDouble(value,false);
398 dvar.setData(id);
399 if(dvarUsedByLaterCycle.contains(dvName)){
400 varCycleValueMap.get(dvName).put(model, id);
401 }else if (dvarTimeArrayUsedByLaterCycle.contains(dvName)){
402 if (varTimeArrayCycleValueMap.containsKey(dvName)){
403 varTimeArrayCycleValueMap.get(dvName).put(model, dvar.data);
404 }else{
405 Map<String, IntDouble> cycleValue = new HashMap<String, IntDouble>();
406 cycleValue.put(model, dvar.data);
407 varTimeArrayCycleValueMap.put(dvName, cycleValue);
408 }
409 }
410 if (varCycleIndexList.contains(dvName) || dvarTimeArrayCycleIndexList.contains(dvName)){
411 if (varCycleIndexValueMap.containsKey(dvName)){
412 varCycleIndexValueMap.get(dvName).put(model, dvar.data);
413 }else{
414 Map<String, IntDouble> cycleValue = new HashMap<String, IntDouble>();
415 cycleValue.put(model, dvar.data);
416 varCycleIndexValueMap.put(dvName, cycleValue);
417 }
418 }
419 String entryNameTS=DssOperation.entryNameTS(dvName, ControlData.timeStep);
420 DataTimeSeries.saveDataToTimeSeries(dvName, entryNameTS, value, dvar);
421 if (timeArrayDvList.contains(dvName)){
422 entryNameTS=DssOperation.entryNameTS(dvName+"__fut__0", ControlData.timeStep);
423 DataTimeSeries.saveDataToTimeSeries(entryNameTS, value, dvar, 0);
424 }
425 }
426
427 if (ControlData.showRunTimeMessage) {
428 System.out.println("Objective Value: "+ControlData.clp_cbc_objective);
429 System.out.println("Assign Dvar Done.");
430 }
