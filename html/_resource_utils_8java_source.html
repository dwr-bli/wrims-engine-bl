10 /**
11 * Get the relative path from one file to another, specifying the directory separator.
12 * If one of the provided resources does not exist, it is assumed to be a file unless it ends with '/' or
13 * '\'.
14 *
15 * @param targetPath targetPath is calculated to this file
16 * @param basePath basePath is calculated from this file
17 * @param pathSeparator directory separator. The platform default is not assumed so that we can test Unix behaviour when running on Windows (for example)
18 * @return
19 */
20 public static String getRelativePath(String targetPath, String basePath, String pathSeparator) {
21
22 // Normalize the paths
23 String normalizedTargetPath = FilenameUtils.normalizeNoEndSeparator(targetPath);
24 String normalizedBasePath = FilenameUtils.normalizeNoEndSeparator(basePath);
25
26 // Undo the changes to the separators made by normalization
27 if (pathSeparator.equals("/")) {
28 normalizedTargetPath = FilenameUtils.separatorsToUnix(normalizedTargetPath);
29 normalizedBasePath = FilenameUtils.separatorsToUnix(normalizedBasePath);
30
31 } else if (pathSeparator.equals("\\")) {
32 normalizedTargetPath = FilenameUtils.separatorsToWindows(normalizedTargetPath);
33 normalizedBasePath = FilenameUtils.separatorsToWindows(normalizedBasePath);
34
35 } else {
36 throw new IllegalArgumentException("Unrecognised dir separator '" + pathSeparator + "'");
37 }
38
39 String[] base = normalizedBasePath.split(Pattern.quote(pathSeparator));
40 String[] target = normalizedTargetPath.split(Pattern.quote(pathSeparator));
41
42 // First get all the common elements. Store them as a string,
43 // and also count how many of them there are.
44 StringBuffer common = new StringBuffer();
45
46 int commonIndex = 0;
47 while (commonIndex < target.length && commonIndex < base.length
48 && target[commonIndex].equalsIgnoreCase(base[commonIndex])) {
49 common.append(target[commonIndex] + pathSeparator);
50 commonIndex++;
51 }
52
53 if (commonIndex == 0) {
54 // No single common path element. This most
55 // likely indicates differing drive letters, like C: and D:.
56 // These paths cannot be relativized.
57 throw new PathResolutionException("No common path element found for '" + normalizedTargetPath + "' and '" + normalizedBasePath
58 + "'");
59 }
60
61 // The number of directories we have to backtrack depends on whether the base is a file or a dir
62 // For example, the relative path from
63 //
64 // /foo/bar/baz/gg/ff to /foo/bar/baz
65 //
66 // ".." if ff is a file
67 // "../.." if ff is a directory
68 //
69 // The following is a heuristic to figure out if the base refers to a file or dir. It's not perfect, because
70 // the resource referred to by this path may not actually exist, but it's the best I can do
71 boolean baseIsFile = true;
72
73 File baseResource = new File(normalizedBasePath);
74
75 if (baseResource.exists()) {
76 baseIsFile = baseResource.isFile();
77
78 } else if (basePath.endsWith(pathSeparator)) {
79 baseIsFile = false;
80 }
81
82 StringBuffer relative = new StringBuffer();
83
84 if (base.length != commonIndex) {
85 int numDirsUp = baseIsFile ? base.length - commonIndex - 1 : base.length - commonIndex;
86
87 for (int i = 0; i < numDirsUp; i++) {
88 relative.append(".." + pathSeparator);
89 }
90 }
91 relative.append(normalizedTargetPath.substring(common.length()));
92 return relative.toString();
93 }
