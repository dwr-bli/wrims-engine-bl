259 public void procOVFile(){
260 ovPartBC=new HashMap<String, String>();
261 File ovFile = new File (ControlData.ovFile);
262 if (!ovFile.exists()){
263 System.out.println("Output variable file doesn't exist. All the timeseries will be written to the csv file.");
264 ControlData.ovOption=0;
265 return;
266 }
267 try {
268 FileInputStream fs = new FileInputStream(ovFile.getAbsolutePath());
269 BufferedReader br = new BufferedReader(new InputStreamReader(fs));
270 String line=br.readLine();
271 if (br == null) {
272 System.out.println("Output variable file doesn't contain data. All the timeseries will be written to the csv file.");
273 };
274 while((line=br.readLine()) !=null){
275 line=line.replace(" ", "").replace("\t", "").toUpperCase();
276 if (line.equals("")) return;
277 String[] parts = line.split(",");
278 if (parts.length>=2){
279 ovPartBC.put(parts[0], parts[1]);
280 }
281 }
282 br.close();
283 fs.close();
284 } catch (FileNotFoundException e) {
285 e.printStackTrace();
286 System.out.println("Output variable file doesn't exist. All the timeseries will be written to the csv file.");
287 ControlData.ovOption=0;
288 } catch (IOException e) {
289 e.printStackTrace();
290 System.out.println("Output variable file has errors. All the timeseries will be written to the csv file.");
291 ControlData.ovOption=0;
