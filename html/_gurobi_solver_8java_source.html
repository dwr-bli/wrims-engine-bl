225 Set multCollection = multMap.keySet();
226 Iterator multIterator = multCollection.iterator();
227 GRBLinExpr expr = new GRBLinExpr();

339 private void checkStatus() throws GRBException {
340 int status = model.get(GRB.IntAttr.Status);
341 if (status == GRB.Status.UNBOUNDED) {
342 System.out.println("The model cannot be solved "
343 + "because it is unbounded");
344 return;
345 }
346 if (status == GRB.Status.OPTIMAL) {
347 System.out.println("The optimal objective is " +
348 model.get(GRB.DoubleAttr.ObjVal));
349 return;
350 }
351 if (status != GRB.Status.INF_OR_UNBD &&
352 status != GRB.Status.INFEASIBLE ) {
353 System.out.println("Optimization was stopped with status " + status);
354 return;
355 }
356
357 // do IIS
358 System.out.println("The model is infeasible; computing IIS");
359 LinkedList<String> removed = new LinkedList<String>();
360
361 // Loop until we reduce to a model that can be solved
362 while (true) {
363 model.computeIIS();
364 System.out.println("\nThe following constraint cannot be satisfied:");
365 for (GRBConstr c : model.getConstrs()) {
366 if (c.get(GRB.IntAttr.IISConstr) == 1) {
367 System.out.println(c.get(GRB.StringAttr.ConstrName));
368 // Remove a single constraint from the model
369 removed.add(c.get(GRB.StringAttr.ConstrName));
370 model.remove(c);
371 break;
372 }
373 }
374
375 System.out.println();
376 model.optimize();
377 status = model.get(GRB.IntAttr.Status);
378
379 if (status == GRB.Status.UNBOUNDED) {
380 System.out.println("The model cannot be solved "
381 + "because it is unbounded");
382 return;
383 }
384 if (status == GRB.Status.OPTIMAL) {
385 break;
386 }
387 if (status != GRB.Status.INF_OR_UNBD &&
388 status != GRB.Status.INFEASIBLE ) {
389 System.out.println("Optimization was stopped with status " +
390 status);
391 return;
392 }
393 }
394
395 System.out.println("\nThe following constraints were removed "
396 + "to get a feasible LP:");
397 for (String s : removed) {
398 System.out.print(s + " ");
399 }
400 System.out.println();
